#include "stdafx.h"
#include "string.h"
#include "stdlib.h"


int _tmain()
{
	FILE *fp, *fp1;
	char lat[20];
	char lon[20];
	char time[30];
	char *a;
	int len;
	fp1 = fopen("d:\\GPX\\export.csv", "w");
	if ((fp = fopen("d:\\GPX\\export.gpx", "r")) == 0)
	{
		printf("Error!\n");
		
		return (-1);
	}
		fseek(fp, 0, SEEK_END);
		len = ftell(fp);
		a = (char *)malloc(len);
		fseek(fp, 0, SEEK_SET);
		fread(a, 1, len, fp);
	    int i = 0;
		fprintf(fp1, "纬度,经度,时间\n");
		while (!(a[i] == '<'&&a[i + 1] == '/'&&a[i + 2] == 'g'&&a[i + 3] == 'p'&&a[i + 4] == 'x'&&a[i + 5] == '>'))
		{
			if (a[i] == ' '&&a[i + 1] == 'l'&&a[i + 2] == 'a'&&a[i + 3] == 't')
			{
				strncpy(lat, &a[i + 6], 9);
				strncpy(lon, &a[i + 22], 10);
				strncpy(time, &a[i + 44], 19);
				lat[9] = '\0';
				lon[10] = '\0';
				time[19] = '\0';
				fprintf(fp1, "%s,%s,%s\n", lat, lon, time);
				
			}
			i++;
		}
	fclose(fp);
	fclose(fp1);
	free(a);
	return 0;
}

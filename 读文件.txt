// ConsoleApplication3.cpp : 定义控制台应用程序的入口点。
//

#include "stdafx.h"
#include "stdio.h"
#include "stdlib.h"


int main()
{

	FILE *fp;
	char ch;
	fp = fopen("d://GPX//export.gpx", "r");
	if (fp == NULL)
	{
		printf("File open error!");
		
	}
	else
	{
		ch = fgetc(fp);
		while (ch != EOF)
		{
			printf("%c", ch);
			ch = fgetc(fp);
		}
		fclose(fp);
	}
	system("pause");
    return 0;
}
